  -H "Authorization: token $GITHUB_TOKEN" \
  https://api.github.com/repos/$OWNER/$REPO/actions/workflows \
  | python3 -c "
import sys, json

    print(f\"  {w['id']:10}  {w['name']:30}  {w['state']}\")"
