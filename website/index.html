Kurulum

git clone <repo-url>
cd hermes_projesi
python -m venv venv
venv\Scripts\activate
pip install -r requirements.txt
python reyment.py doctor

Kullanim

# Ajani calistir
python start.py

# Sadece Web UI
python start.py --dashboard-only

# CLI ile hedef ver
python reyment.py run "masaustunde test.txt olustur"

# Gateway (Telegram + WebSocket)
python reyment.py gateway start
